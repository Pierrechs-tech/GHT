import requests

url = "https://chat.openai.com/sandbox/mnt/data/hopitaux-theme.zip"
output_path = "hopitaux-theme.zip"

response = requests.get(url, stream=True)
response.raise_for_status()  # Stoppe si erreur HTTP

with open(output_path, "wb") as f:
    for chunk in response.iter_content(chunk_size=8192):
        if chunk:
            f.write(chunk)

print("Téléchargement terminé :", output_path)
